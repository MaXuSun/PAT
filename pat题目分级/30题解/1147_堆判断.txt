#include <iostream>
#include <algorithm>
#include <vector>
#define MAXN 1010
using namespace std;
int H[MAXN];
int n,m;
vector<int> post;
bool isbig=true;
bool issmall = true;

void checkHeap(){               //ºÏ≤È «ƒƒ÷÷∂—
    for(int i = 2;i <= n;i++){
        if(H[i/2]>H[i])issmall=false;
        if(H[i/2]<H[i])isbig=false;
    }
}

void postOrder(int s){
    if(s > n)return;
    postOrder(s*2);
    postOrder(s*2+1);
    post.push_back(H[s]);
}

int main(){
    cin >> m >> n;
    for(int j = 0;j < m;j++){
        for(int i = 1;i <= n;i++){
            cin >> H[i];
        }

        checkHeap();
        if(isbig)printf("Max Heap\n");
        else if(issmall)printf("Min Heap\n");
        else printf("Not Heap\n");

        postOrder(1);
        for(int i = 0;i < post.size();i++){
            if(i!=0)printf(" ");
            printf("%d",post[i]);
        }
        printf("\n");
        issmall=true,isbig=true,post.clear();
    }

    return 0;
}
